11class TempdirFactory(object): 

12 """Factory for temporary directories under the common base temp directory. 

13 

14 The base directory can be configured using the ``--basetemp`` option. 

15 """

17 def __init__(self, config): 

18 self.config = config 

19 self.trace = config.trace.get("tmpdir")

21 def ensuretemp(self, string, dir=1): 

22 """ (deprecated) return temporary directory path with 

23 the given string as the trailing part. It is usually 

24 better to use the 'tmpdir' function argument which 

25 provides an empty unique-per-test-invocation directory 

26 and is guaranteed to be empty. 

27 """ 

28 # py.log._apiwarn(">1.1", "use tmpdir function argument") 

29 return self.getbasetemp().ensure(string, dir=dir) 

30 

31 def mktemp(self, basename, numbered=True): 

32 """Create a subdirectory of the base temporary directory and return it. 

33 If ``numbered``, ensure the directory is unique by adding a number 

34 prefix greater than any existing one. 

35 """ 

36 basetemp = self.getbasetemp() 

37 if not numbered: 

38 p = basetemp.mkdir(basename) 

39 else: 

40 p = py.path.local.make_numbered_dir(prefix=basename, 

41 keep=0, rootdir=basetemp, lock_timeout=None) 

42 self.trace("mktemp", p) 

43 return p

45 def getbasetemp(self): 

46 """ return base temporary directory. """ 

47 try: 

48 return self._basetemp 

49 except AttributeError: 

50 basetemp = self.config.option.basetemp 

51 if basetemp: 

52 basetemp = py.path.local(basetemp) 

53 if basetemp.check(): 

54 basetemp.remove() 

55 basetemp.mkdir() 

56 else: 

57 temproot = py.path.local.get_temproot() 

58 user = get_user() 

59 if user: 

60 # use a sub-directory in the temproot to speed-up 

61 # make_numbered_dir() call 

62 rootdir = temproot.join('pytest-of-%s' % user) 

63 else: 

64 rootdir = temproot 

65 rootdir.ensure(dir=1) 

66 basetemp = py.path.local.make_numbered_dir(prefix='pytest-', 

67 rootdir=rootdir) 

68 self._basetemp = t = basetemp.realpath() 

69 self.trace("new basetemp", t) 

70 return t

93

98 mp = MonkeyPatch()
